\def\@lesson{}
\newcommand{\lesson}[3]{
  \setcounter{section}{#1}
  \renewcommand\thesubsection{#1.\arabic{subsection}}

  \ifthenelse{\isempty{#3}}{
    \def\@lesson{Lecture #1}
  }{
    \def\@lesson{Lecture #1: #3}
  }
  \testdateparts{#2}
  \hrule
  \vspace*{-0.3cm}
  \section*{\@lesson}
  \addcontentsline{toc}{section}{\@lesson}
}

}
