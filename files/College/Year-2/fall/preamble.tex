\def\@lesson{}
\newcommand{\lesson}[3]{
  \renewcommand\thesubsection{#1.\arabic{subsection}}

  \ifthenelse{\isempty{#3}}{
    \def\@lesson{Lecture #1}
  }{
    \def\@lesson{Lecture #1: #3}
  }
  \testdateparts{#2}
  \hrule
  \vspace*{-0.3cm}
  \section*{\@lesson}
  \addcontentsline{toc}{section}{\@lesson}
}

\usepackage{thmtools}
\usepackage[framemethod=TikZ]{mdframed}
\mdfsetup{skipabove=1em,skipbelow=0em}

\theoremstyle{definition}
\newcommand{\createnewtheorem}[2]{
  \declaretheoremstyle[
  headfont=\bfseries\sffamily\color{#2!70!black}, bodyfont=\normalfont,
  mdframed={
    linewidth=2pt,
    rightline=false, topline=false, bottomline=false,
    linecolor=#2, backgroundcolor=#2!5,
  },
  ]{#1}
}

\createnewtheorem{thmgreenbox}{ForestGreen}
\createnewtheorem{thmbluebox}{NavyBlue}
\createnewtheorem{thmredbox}{RawSienna}

\declaretheorem[style=thmgreenbox, name=Definition]{definition}
\declaretheorem[style=thmredbox, name=Theorem]{theorem}
\declaretheorem[style=thmbluebox, numbered=no, name=Example]{example}
