\begin{problem}[2.3]
  Let $(X, \metric)$ be a metric space and $A \subset X$. Use Proposition 2.6 to prove that
  \[%
    \partial A = \partial(X \setminus A) = \bar{A} \cap \overline{X \setminus A}
  .\]%
\end{problem}

\begin{solution}
  Assume $x \in \partial A$. By definition, that means $x \in \bar{A} \setminus \sint A$. Then, $x \in \bar{A}$ and $x \notin \sint A$. By Proposition 2.6, $x \in \sint A$ is equivalent to $x \in \overline{X \setminus A}$. Therefore, $x \in \bar{A}$ and $x \in \overline{X \setminus A}$, which implies $x \in \bar{A} \cap \overline{X \setminus A}$. Hence, $\partial A \subset \bar{A} \cap \overline{X \setminus A}$.

  Now, for the converse. Assume $x \in \bar{A} \cap \overline{X \setminus A}$. Then, $x \in \bar{A}$ and $x \in \overline{X \setminus A}$. By Proposition 2.6, $x \in \overline{X \setminus A}$ implies $x \notin \sint A$. Therefore, $x \in \bar{A}$ and $x \notin \sint A$, which means $x \in \partial A$. Hence, $\bar{A} \cap \overline{X \setminus A} \subset \partial A$.

  Combining both directions, we conclude that $\partial A = \bar{A} \cap \overline{X \setminus A}$.
\end{solution}
